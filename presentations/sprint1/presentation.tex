\section{Introduction}
\frame{
	\frametitle{What is PHARC?}
	Plastic Surgeons need to organize lots of pictures.
	However, they don't have good software:
	\begin{itemize}
	\item Need to organize around patients, diagnoses, treatments
	\item Have particular interface requirements
	\item Need particular storage requirements
	\end{itemize}
	Client asked us to build special application
}

\section{Project Plan}
\frame{
	\frametitle{Basic Client Requirements}
	\begin{itemize}
	\item Search archive by patient, treatment, diagnosis, physician, etcetera
	\item Easy interoperability with email, photo editing, and powerpoint
	\item Transparent filesystem layout. No opaque databases!
	\end{itemize}
}

\frame{
	\frametitle{Basic Implementation}
	\begin{itemize}
	\item Building a local application
	\item Using Python and Qt for cross-platform functionality
	\item Open-source using git and github
	\item Modular design to help with scheduling
	\end{itemize}
}

\section{Storage}
\frame{
	\frametitle{Storage Requirements}
	\begin{itemize}
	\item Multiple-user simultaneous access
	\item Works over network drives
	\item User can access archive without using application
	\end{itemize}
}

\frame{
	\frametitle{Storage Architecture}
	\begin{itemize}
	\item Our solution: use the filesystem. \begin{itemize}
		\item Organize photos into directories by patient and session
		\item Store notes and other information in text files
	\end{itemize}
	\item This is slow, though. Solution: database cache \begin{itemize}
		\item Cache information in local database
		\item Maintain separate database per user
		\item Also makes network operation bearable
	\end{itemize}
	\end{itemize}
}

\section{Interface}
\frame{
	\frametitle{Interface Implementation}
	\begin{itemize}
		\item Framework: Qt \begin{itemize}
			\item Popular, well-supported
			\item Powerful
			\item Cross-platform
		\end{itemize}
		\item Python doesn't do threading, so processes instead
	\end{itemize}
}

\frame{
	\frametitle{Interface Design}
	\begin{itemize}
		\item Import window
		\item Main window for searching/selecting patients
		\item Patient Detail window for viewing/editing patient information
		\item Image Edit window for lightweight editing operations
	\end{itemize}
}
